# 产品配置算法 - 简化版伪代码流程

> **版本**：2026-07-09 简化版（全部按独栋处理，暂不处理双拼转换）
> **目标**：根据规划用地指标自动计算各类厂房和配套建筑的配置
> **校验重点**：流程完整性、边界条件、数据流一致性、公式正确性

---

## 一、全局变量与输入

### 输入参数
- `S` = 规划用地面积
- `F` = 容积率
- `D` = 建筑密度（F<1.5→45%, 1.5≤F<2.0→42%, F≥2.0→40%）
- `R1` = 科研办公用房占比
- `R2` = 配套用房占比
- `Hl` = 限高
- `selectedProducts` = Set，用户已选产品类型
- `productOptions` = 各产品类型的楼层和面积段配置

### 派生变量
- `targetBase = S * D`
- `targetCap = S * F`
- `targetRdArea = targetCap * R1`
- `targetAncillaryArea = targetCap * R2`

## 二、主流程

```
function calculateProductConfig(projectData, selectedProducts, productOptions):

   // ===== 第零步：前置处理 =====
   result0 = step0_preprocess(selectedProducts, productOptions, Hl)
   selectedProducts = result0.selectedProducts
   productOptions = result0.productOptions
   layerFloors = result0.layerFloors          // 降级后的分层层数数组
   splitFloors = result0.splitFloors          // 分栋层数数组
   referenceHeight = result0.referenceHeight  // 宿舍参考高度上限
   layerDowngraded = result0.layerDowngraded  // 是否因限高降级（用于备注）

   // ===== 步骤一：产业大厦 =====
   fixedResult = step1_tower(selectedProducts, productOptions, targetRdArea, Hl)
   allConfigs = fixedResult.configs
   fixedBase = fixedResult.base
   fixedCap = fixedResult.cap

   // ===== 步骤二：配套用房 =====
   ancillaryResult = step2_ancillary(selectedProducts, productOptions,
                                     targetAncillaryArea, referenceHeight)
   allConfigs = merge(allConfigs, ancillaryResult.configs)
   fixedBase += ancillaryResult.base
   fixedCap += ancillaryResult.cap

   // ===== 步骤三：厂房预处理 =====
   factoryResult = step3_factory_preprocess(selectedProducts, productOptions,
                                            layerFloors, splitFloors,
                                            targetBase, targetCap,
                                            fixedBase, fixedCap)
   fixedFactoryConfigs = factoryResult.fixedConfigs
   bhConfigs = factoryResult.bhConfigs
   blConfigs = factoryResult.blConfigs
   remainingBase = factoryResult.remainingBase
   remainingCap = factoryResult.remainingCap

   // ===== 步骤四：Bh/Bl核心算法 =====
   coreResult = step4_core_algorithm(bhConfigs, blConfigs,
                                     remainingBase, remainingCap)
   finalCounts = coreResult.counts
   activeBhConfigs = coreResult.bhConfigs
   activeBlConfigs = coreResult.blConfigs
   isSingleType = coreResult.isSingleType

   // ===== 步骤五：外层循环优化 =====
   outerResult = step5_outer_loop(finalCounts,
                                  activeBhConfigs, activeBlConfigs,
                                  fixedFactoryConfigs,
                                  remainingBase, remainingCap,
                                  targetBase, targetCap,
                                  fixedBase, fixedCap,
                                  isSingleType, S)

   // ===== 输出合并 =====
   output = allConfigs
   for each config in fixedFactoryConfigs:
      if config.count > 0: output.push(config)
   for each config in outerResult.bhConfigs + outerResult.blConfigs:
      if config.count > 0: output.push(config)

   return output
```

## 三、第零步：前置处理

```
function step0_preprocess(selectedProducts, productOptions, Hl):

   layerFloors = []
   splitFloors = []
   downgradedFloors = []
   layerDowngraded = false

   // 0.1 分层厂房层数降级
   if selectedProducts 包含 'layer':
      for floor in productOptions['layer'].floors:
         currentFloor = floor
         totalH = 7.2 + 5.1 + 4.5 * (currentFloor - 2) + 1.2
         while totalH > Hl and currentFloor > 1:
            currentFloor--
            totalH = 7.2 + 5.1 + 4.5 * (currentFloor - 2) + 1.2
         if totalH <= Hl:
            if currentFloor >= 5:
               layerFloors.push(currentFloor)
               if currentFloor < floor: layerDowngraded = true
            else if currentFloor <= 4:
               downgradedFloors.push(currentFloor)
      if layerFloors 为空:
         selectedProducts 删除 'layer'

   // 0.2 分栋厂房层数降级
   if selectedProducts 包含 'split':
      for floor in productOptions['split'].floors:
         if floor == 3.5:
            totalH = 22.5 + 1.2
         else:
            totalH = 7.2 + 5.1 + 4.5 * (floor - 2) + 1.2
         if totalH <= Hl:
            splitFloors.push(floor)
      if splitFloors 为空:
         selectedProducts 删除 'split'

   // 0.3 分层降级为分栋的合并处理
   if downgradedFloors 非空:
      selectedProducts 删除 'layer'
      if selectedProducts 不包含 'split':
         selectedProducts 添加 'split'

      // 原始分层面积段直接合并到分栋
      originalLayerAreas = productOptions['layer'].areas
      convertedAreas = []
      for area in originalLayerAreas:
         convertedAreas.push(area)  // 简化规则：不再还原双拼

      existingSplitAreas = productOptions['split'].areas 或 []
      mergedAreas = 去重排序(existingSplitAreas + convertedAreas)
      productOptions['split'].areas = mergedAreas

      convertedFloors = 去重排序(downgradedFloors)
      existingSplitFloors = productOptions['split'].floors 或 []
      mergedFloors = 去重排序(existingSplitFloors + convertedFloors)
      productOptions['split'].floors = mergedFloors

   // 0.4 更新 productOptions 中的分层参数
   if selectedProducts 包含 'layer' 且 layerFloors 非空:
      productOptions['layer'].floors = layerFloors
      // 面积段保持原始值，不转换

   // 0.5 配套宿舍参考高度上限
   referenceHeight = Hl
   if selectedProducts 包含 'tower':
      // 产业大厦在步骤一中才计算，这里先占位
      referenceHeight = null  // 后续用 towerCfg.totalHeight 回填
   else if layerFloors 非空:
      referenceHeight = max(layerFloors.map(f => 7.2 + 5.1 + 4.5 * (f - 2) + 1.2))

   return {
      selectedProducts, productOptions,
      layerFloors, splitFloors, referenceHeight, layerDowngraded
   }
```

function step1_tower(selectedProducts, productOptions, targetRdArea, Hl):

   configs = []
   base = 0
   cap = 0

   if selectedProducts 不包含 'tower' 或 targetRdArea <= 0:
      return { configs, base, cap }

   areaRef = productOptions['tower'].areas[0]

   floors = max(3, ceil(targetRdArea / areaRef))
   totalH = 6.6 + 4.5 * (floors - 2) + 4.5 + 1.2

   if totalH > Hl:
      floors = max(3, floor((Hl - 6.6 - 4.5 - 1.2) / 4.5) + 2)
      totalH = 6.6 + 4.5 * (floors - 2) + 4.5 + 1.2

   exactBase = targetRdArea / floors
   finalBase = floor(exactBase / 50) * 50
   actualArea = finalBase * floors

   if |actualArea - targetRdArea| / targetRdArea > 0.002:
      finalBase = floor(exactBase / 20) * 20
      actualArea = finalBase * floors

   paxLift = max(1, ceil(actualArea / 4000) - 2)

   configs.push({
      id: 'tower', type: '产业大厦',
      base: finalBase, unitCap: actualArea, floors, totalHeight: totalH,
      count: 1, elevator: paxLift + '客2货'
   })

   base = finalBase
   cap = actualArea

   // 回填 referenceHeight（如果前面是null）
   // 注：主流程中需要处理这个回填

   return { configs, base, cap, towerHeight: totalH }
```

function step2_ancillary(selectedProducts, productOptions,
                         targetAncillaryArea, referenceHeight):

   configs = []
   base = 0
   cap = 0

   if selectedProducts 不包含 'dorm' 且 不包含 'support':
      return { configs, base, cap }

   totalAncillary = targetAncillaryArea

   // 情况A：totalAncillary <= 2400 且选了配套楼
   if totalAncillary <= 2400 且 selectedProducts 包含 'support':
      supportFloors = totalAncillary < 1200 ? 2 : 3
      totalH = supportFloors == 2 ? 12.9 : 18.3
      exactBase = totalAncillary / supportFloors
      supportBase = floor(exactBase / 50) * 50
      finalSupportCap = supportBase * supportFloors
      if |finalSupportCap - totalAncillary| / totalAncillary > 0.002:
         supportBase = floor(exactBase / 20) * 20
         finalSupportCap = supportBase * supportFloors

      configs.push({ id: 'support', type: '配套楼',
                     base: supportBase, unitCap: finalSupportCap,
                     floors: supportFloors, totalHeight: totalH, count: 1 })
      base = supportBase
      cap = finalSupportCap

   // 情况B：同时选了配套楼和配套宿舍
   else if selectedProducts 包含 'dorm' 且 包含 'support':
      // 配套楼分配
      if totalAncillary < 1200:
         supportFloors = 2
         finalSupportCap = totalAncillary
         supportBase = floor(totalAncillary / 2 / 50) * 50
      else if totalAncillary < 2400:
         supportFloors = 3
         finalSupportCap = totalAncillary
         supportBase = floor(totalAncillary / 3 / 50) * 50
      else if totalAncillary < 3600:
         supportFloors = 3
         finalSupportCap = 1200
         supportBase = 400
      else:
         supportFloors = 3
         finalSupportCap = 2400
         supportBase = 800

      totalH = supportFloors == 2 ? 12.9 : 18.3
      configs.push({ id: 'support', type: '配套楼',
                     base: supportBase, unitCap: finalSupportCap,
                     floors: supportFloors, totalHeight: totalH, count: 1 })
      base = supportBase
      cap = finalSupportCap

      // 配套宿舍
      if totalAncillary > finalSupportCap:
         dormCap = totalAncillary - finalSupportCap
         dormResult = calc_dorm(dormCap, referenceHeight)
         configs.push(dormResult.config)
         base += dormResult.base
         cap += dormResult.cap

   // 情况C：仅选了配套宿舍
   else if selectedProducts 包含 'dorm':
      dormResult = calc_dorm(totalAncillary, referenceHeight)
      configs.push(dormResult.config)
      base = dormResult.base
      cap = dormResult.cap

   return { configs, base, cap }


function calc_dorm(dormCap, referenceHeight):

   estFloors = max(2, floor((referenceHeight - 4.8 - 1.2) / 3.6) + 1)
   singleArea = dormCap / estFloors
   if singleArea < 600: base = 600
   else if singleArea > 1200: base = 1200
   else: base = round(singleArea / 100) * 100

   floors = max(2, ceil(dormCap / base))
   totalH = 4.8 + 3.6 * (floors - 2) + 1.2

   if totalH > referenceHeight:
      totalH = 4.8 + 3.3 * (floors - 2) + 1.2
      if totalH > referenceHeight:
         floors = max(2, floor((referenceHeight - 4.8 - 1.2) / 3.3) + 2)
         totalH = 4.8 + 3.3 * (floors - 2) + 1.2

   exactBase = dormCap / floors
   finalBase = floor(exactBase / 20) * 20
   unitArea = finalBase * floors

   config = {
      id: 'dorm', type: '配套宿舍',
      base: finalBase, unitCap: unitArea, floors,
      totalHeight: totalH, count: 1
   }

   return { config, base: finalBase, cap: unitArea }

## 六、步骤三：厂房预处理

```
function step3_factory_preprocess(selectedProducts, productOptions,
                                  layerFloors, splitFloors,
                                  targetBase, targetCap,
                                  fixedBase, fixedCap):

   fixedConfigs = []
   usedBase = 0
   usedCap = 0

   hasLightSteel = selectedProducts 包含 'light-steel'
   hasSplit = selectedProducts 包含 'split'
   hasLayer = selectedProducts 包含 'layer'

   totalLayerTypes = splitFloors.length + layerFloors.length

   // 3.3 FL预处理
   while totalLayerTypes > 2:
      if layerFloors.length > 1:
         preprocessType = 'layer'
         preprocessFloors = layerFloors
      else if splitFloors.length > 1:
         preprocessType = 'split'
         preprocessFloors = splitFloors
      else:
         break

      maxFloor = max(preprocessFloors)
      maxArea = max(productOptions[preprocessType].areas)
      a = maxArea * maxFloor

      if preprocessType == 'layer':
         FL = (S * F) / a / 2
      else:
         FL = (S * F) / a

      if FL < 1: break

      for area in productOptions[preprocessType].areas:
         base = area                    // 简化规则：不再双拼转换
         floors = maxFloor
         totalH = 7.2 + 5.1 + 4.5 * (floors - 2) + 1.2
         unitCap = base * floors
         fixedConfigs.push({
            id: preprocessType, type: 类型名称,
            base, unitCap, floors, totalHeight: totalH,
            count: 1, isFixed: true
         })
         usedBase += base
         usedCap += unitCap

      preprocessFloors = 过滤掉 maxFloor
      if preprocessType == 'layer':
         layerFloors = preprocessFloors
      else:
         splitFloors = preprocessFloors

      totalLayerTypes = splitFloors.length + layerFloors.length

   // 3.4 三种厂房类型处理
   if hasLightSteel 且 hasSplit 且 hasLayer:
      minLayerFloor = min(layerFloors)
      minLayerArea = min(productOptions['layer'].areas)
      base = minLayerArea
      floors = minLayerFloor
      totalH = 7.2 + 5.1 + 4.5 * (floors - 2) + 1.2
      unitCap = base * floors
      fixedConfigs.push({ id: 'layer', type: '分层厂房',
                          base, unitCap, floors, totalHeight: totalH,
                          count: 1, isFixed: true })
      usedBase += base
      usedCap += unitCap
      layerFloors = 过滤掉 minLayerFloor

      // 剩余退化为分栋+轻钢
      if splitFloors.length > 1:
         // FL预处理分栋（同上）
         ...

   // 3.5 创建Bh/Bl配置
   bhConfigs = []
   blConfigs = []

   allTypes = []
   if hasLightSteel: allTypes.push({ type: 'light-steel', floors: [2.1] })
   if hasSplit 且 splitFloors.length > 0: allTypes.push({ type: 'split', floors: splitFloors })
   if hasLayer 且 layerFloors.length > 0: allTypes.push({ type: 'layer', floors: layerFloors })

   if allTypes.length == 0:
      // 无厂房
      pass
   else if allTypes.length == 1:
      t = allTypes[0]
      bhConfigs = createConfigs(t.type, t.floors[0], productOptions)
   else if allTypes.length == 2:
      t1 = allTypes[0]
      t2 = allTypes[1]
      eff1 = t1.type == 'light-steel' ? 2.1 : t1.floors[0]
      eff2 = t2.type == 'light-steel' ? 2.1 : t2.floors[0]
      if eff1 >= eff2:
         bhConfigs = createConfigs(t1.type, t1.floors[0], productOptions)
         blConfigs = createConfigs(t2.type, t2.floors[0], productOptions)
      else:
         bhConfigs = createConfigs(t2.type, t2.floors[0], productOptions)
         blConfigs = createConfigs(t1.type, t1.floors[0], productOptions)

   remainingBase = targetBase - fixedBase - usedBase
   remainingCap = targetCap - fixedCap - usedCap

   return {
      fixedConfigs, bhConfigs, blConfigs,
      remainingBase, remainingCap
   }


function createConfigs(type, floor, productOptions):

   configs = []
   areas = productOptions[type].areas 或 []

   for area in areas:
      base = area
      if type == 'light-steel':
         unitCap = base * 2 + 200
         totalH = 13.2
      else if type == 'split' 且 floor == 3.5:
         unitCap = base * 3.5
         totalH = 22.5 + 1.2
      else:
         unitCap = base * floor
         totalH = 7.2 + 5.1 + 4.5 * (floor - 2) + 1.2

      configs.push({
         id: type, type: 类型名称,
         base, unitCap, floors: floor, totalHeight: totalH
      })

   return configs
```

## 七、步骤四：Bh/Bl核心算法

```
function step4_core_algorithm(bhConfigs, blConfigs, remainingBase, remainingCap):

   if bhConfigs 为空 且 blConfigs 为空:
      return { counts: [], bhConfigs, blConfigs, isSingleType: true }

   effBh = bhConfigs 的平均效率 (unitCap / base)
   effBl = blConfigs 的平均效率
   avgEff = remainingCap / remainingBase
   EPS = 1e-9

   // 4.2 解方程
   if avgEff > effBl + EPS 且 avgEff < effBh - EPS:
      x = (remainingCap - effBl * remainingBase) / (effBh - effBl)
      y = remainingBase - x
      targetCapBh = effBh * x
      targetCapBl = effBl * y
      targetBaseBh = x
      targetBaseBl = y
   else if avgEff >= effBh - EPS:
      targetCapBh = remainingCap
      targetCapBl = 0
      targetBaseBh = remainingBase
      targetBaseBl = 0
   else if avgEff <= effBl + EPS:
      targetCapBh = 0
      targetCapBl = remainingCap
      targetBaseBh = 0
      targetBaseBl = remainingBase
   else:
      x = remainingBase / 2
      y = remainingBase / 2
      targetCapBh = effBh * x
      targetCapBl = effBl * y
      targetBaseBh = x
      targetBaseBl = y

   // 4.3 整数方程求解
   bhResult = solveIntegerEquation(targetCapBh, bhConfigs, targetBaseBh * 1.05)
   blResult = solveIntegerEquation(targetCapBl, blConfigs, targetBaseBl * 1.05)

   finalCounts = [...bhResult.counts, ...blResult.counts]

   // 4.5 取消判断
   bhCount = sum(finalCounts[0..bhConfigs.length-1])
   blCount = sum(finalCounts[bhConfigs.length..])

   if bhCount == 0 且 blCount > 0:
      finalCounts = solveSingleType(blConfigs, remainingCap, remainingBase)
      activeBhConfigs = []
      activeBlConfigs = blConfigs
      isSingleType = true
   else if blCount == 0 且 bhCount > 0:
      finalCounts = solveSingleType(bhConfigs, remainingCap, remainingBase)
      activeBhConfigs = bhConfigs
      activeBlConfigs = []
      isSingleType = true
   else:
      activeBhConfigs = bhConfigs
      activeBlConfigs = blConfigs
      isSingleType = false

   return { counts: finalCounts,
            bhConfigs: activeBhConfigs, blConfigs: activeBlConfigs,
            isSingleType }


function solveSingleType(configs, targetCap, targetBase):
   result = solveIntegerEquation(targetCap, configs, targetBase * 1.05)
   return result.counts


function solveIntegerEquation(targetCap, configs, targetBaseLimit):

   hasBaseLimit = targetBaseLimit != null 且 > 0
   k = configs.length

   if k == 0:
      return { counts: [], totalCap: 0, totalBase: 0 }

   if k == 1:
      c = configs[0]
      n = max(1, round(targetCap / c.unitCap))
      bestN = n
      bestScore = infinity
      startN = max(1, n - 200)
      endN = n + 200
      for testN from startN to endN:
         testCap = c.unitCap * testN
         testBase = c.base * testN
         if hasBaseLimit 且 testBase > targetBaseLimit * 1.05: continue
         score = |testCap - targetCap| + |testBase - targetBaseLimit| * 2
         if score < bestScore:
            bestScore = score
            bestN = testN
      return { counts: [bestN],
               totalCap: c.unitCap * bestN,
               totalBase: c.base * bestN }

   // 多配置 DFS
   best = null
   bestScore = infinity

   getSearchRange(targetVal, unitVal):
      estN = max(0, round(targetVal / unitVal))
      if estN <= 3: return 4
      if estN <= 10: return 8
      if estN <= 30: return 15
      return 30

   dfs(idx, current, currentCap, currentBase):
      if idx == k - 1:
         c = configs[idx]
         n = max(0, round((targetCap - currentCap) / c.unitCap))
         range = getSearchRange(targetCap - currentCap, c.unitCap)
         for dn from -range to range:
            nn = max(0, n + dn)
            tc = currentCap + c.unitCap * nn
            tb = currentBase + c.base * nn
            if hasBaseLimit 且 tb > targetBaseLimit * 1.05: continue
            if hasBaseLimit 且 currentBase > targetBaseLimit * 1.2: return
            if currentCap > targetCap * 1.5: return

            capDiff = |tc - targetCap|
            baseDiff = hasBaseLimit ? |tb - targetBaseLimit| : 0
            allCounts = [...current, nn]
            nonZeroCount = 非零个数
            penalty = (k >= 2 且 nonZeroCount < 2) ? 50000 : 0
            score = capDiff + baseDiff * 2 + penalty

            if score < bestScore:
               bestScore = score
               best = [...current, nn]
         return

      c = configs[idx]
      n = max(0, round((targetCap - currentCap) / c.unitCap))
      range = getSearchRange(targetCap - currentCap, c.unitCap)
      for dn from -range to range:
         nn = max(0, n + dn)
         nextBase = currentBase + c.base * nn
         nextCap = currentCap + c.unitCap * nn
         if hasBaseLimit 且 nextBase > targetBaseLimit * 1.2: continue
         if nextCap > targetCap * 1.5: continue
         dfs(idx + 1, [...current, nn], nextCap, nextBase)

   dfs(0, [], 0, 0)

   if !best:
      // fallback：用最小面积段
      sortedConfigs = configs 按 base 排序
      minBase = sortedConfigs[0].base
      maxN = hasBaseLimit ? floor(targetBaseLimit * 1.05 / minBase) : 100
      if maxN <= 0:
         return { counts: configs.map(() => 0), totalCap: 0, totalBase: 0 }

      bestN = max(1, min(maxN, round(targetCap / sortedConfigs[0].unitCap)))
      bestScore = infinity
      for n from 1 to maxN:
         tc = sortedConfigs[0].unitCap * n
         tb = sortedConfigs[0].base * n
         score = |tc - targetCap| + |tb - targetBaseLimit| * 2
         if score < bestScore:
            bestScore = score
            bestN = n

      result = configs.map(() => 0)
      minIdx = configs 中 sortedConfigs[0] 的索引
      result[minIdx] = bestN
      return { counts: result,
               totalCap: sortedConfigs[0].unitCap * bestN,
               totalBase: sortedConfigs[0].base * bestN }

   totalCap = sum(c.unitCap * best[i])
   totalBase = sum(c.base * best[i])
   return { counts: best, totalCap, totalBase }
```

## 八、步骤五：外层循环

```
function step5_outer_loop(finalCounts, bhConfigs, blConfigs,
                          fixedFactoryConfigs,
                          remainingBase, remainingCap,
                          targetBase, targetCap,
                          fixedBase, fixedCap,
                          isSingleType, landArea):

   counts = [...finalCounts]
   currentBhConfigs = [...bhConfigs]
   currentBlConfigs = [...blConfigs]
   oscillationHistory = []

   CAP_LO = targetCap * 0.995
   CAP_HI = targetCap * 1.005

   for round from 0 to 9:

      totals = calcTotals(counts, currentBhConfigs, currentBlConfigs, fixedFactoryConfigs)
      totals.base += fixedBase
      totals.cap += fixedCap

      // 振荡检测
      densityDiff = totals.base - targetBase
      if densityDiff > targetBase * 0.05: oscillationHistory.push(1)
      else if densityDiff < -targetBase * 0.05: oscillationHistory.push(-1)
      else: oscillationHistory.push(0)
      if oscillationHistory.length > 5: oscillationHistory.shift()

      // 分布检查
      allConfigs = [...currentBhConfigs, ...currentBlConfigs]
      distCheck = checkDistribution(counts, allConfigs)

      if !distCheck.needsOptimization 且 !detectOscillation(oscillationHistory):
         if totals.cap >= CAP_LO 且 totals.cap <= CAP_HI:
            break

      // 需要优化
      if distCheck.needsOptimization 或 detectOscillation(oscillationHistory):
         type = distCheck.type
         a = distCheck.a
         b = distCheck.b
         eff = distCheck.eff

         typeIndices = allConfigs 中 id == type 的索引
         S_type = sum(allConfigs[i].base * counts[i])  // i in typeIndices

         if isSingleType:
            targetConfigs = (type == 'layer') ? currentBhConfigs : currentBlConfigs
            enrichedConfigs = enrichConfigs(targetConfigs, landArea)
            newCounts = solveIntegerEquation(remainingCap, enrichedConfigs, remainingBase * 1.05)

            if type == 'layer': currentBhConfigs = enrichedConfigs
            else: currentBlConfigs = enrichedConfigs
            counts = newCounts.counts

            totals = calcTotals(counts, currentBhConfigs, currentBlConfigs, fixedFactoryConfigs)
            totals.base += fixedBase
            totals.cap += fixedCap

            distCheck2 = checkDistribution(counts, [...currentBhConfigs, ...currentBlConfigs])
            if distCheck2.needsOptimization 或 detectOscillation(oscillationHistory):
               optimized = manualDistribute(S_type, a, b, eff)
               counts = mapOptimizedToCounts(optimized, counts, currentBhConfigs, currentBlConfigs, type)
         else:
            optimized = manualDistribute(S_type, a, b, eff)
            counts = mapOptimizedToCounts(optimized, counts, currentBhConfigs, currentBlConfigs, type)

      // 精细调节
      fineTuned = fineTune(counts, [...currentBhConfigs, ...currentBlConfigs],
                           targetCap, targetBase, fixedBase, fixedCap)
      counts = fineTuned.counts
      totals = fineTuned.totals

      if totals.cap >= CAP_LO 且 totals.cap <= CAP_HI 且 |totals.base - targetBase| <= targetBase * 0.05:
         break

   return { counts, bhConfigs: currentBhConfigs, blConfigs: currentBlConfigs }


function calcTotals(counts, bhConfigs, blConfigs, fixedFactoryConfigs):
   base = 0; cap = 0
   for c in fixedFactoryConfigs:
      base += c.base * c.count
      cap += c.unitCap * c.count
   allConfigs = [...bhConfigs, ...blConfigs]
   for i from 0 to counts.length - 1:
      if allConfigs[i] 且 counts[i] > 0:
         base += allConfigs[i].base * counts[i]
         cap += allConfigs[i].unitCap * counts[i]
   return { base, cap }
