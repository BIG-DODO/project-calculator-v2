# 产品配置算法 - 完整伪代码流程（第八轮修订版）

> **版本**：2026-07-10 第八轮修订版（B2 动态 Bh/Bl 修正）
> **核心修复**：
> 1. FL 预处理规则修正为按 4/6/8 阈值分配，避免过度固定高层数产品
> 2. FL 预处理目标修正为「所有厂房效率类型 ≤ 2」，而非仅 split/layer 层数类型
> 3. 新增「分栋 + 轻钢」组合的专门处理（方案 B）
> 4. 新增「四层分栋 + 六层分层」组合的密度调整处理（方案 A）
> 5. 取消规则限制：仅允许同类型产品间取消 Bh/Bl 归一；不同产品类型必须保留，必要时通过密度微调或最终抛弃并提示用户
> 6. **B2 分支**：轻钢与分栋的 Bh/Bl 不再固定，按实际平均效率动态分配

## 一、常量与工具函数

```js
TYPE_NAMES = {
  'light-steel': '轻钢厂房',
  'split': '分栋厂房',
  'layer': '分层厂房',
  'tower': '产业大厦',
  'support': '配套楼',
  'dorm': '配套宿舍'
}

function getProductName(id):
  return TYPE_NAMES[id] || id

function calcDensity(far):
  if far < 1.5: return 0.45
  if far < 2.0: return 0.42
  return 0.40

function computeUnitCap(type, floor, base):
  if type == 'light-steel':
    return base * 2 + 200
  if type == 'split' and floor == 3.5:
    return base * 3.5
  return base * floor

function computeTotalHeight(type, floor, standardHeight = 4.5):
  if type == 'light-steel':
    return 13.2
  if floor == 3.5:
    return 22.5  // 与 4 层等价
  if type == 'tower':
    return 6.6 + standardHeight * (floor - 1) + 1.2
  if type == 'dorm':
    return 4.8 + 3.6 * (floor - 1) + 1.2
  if floor == 1:
    return 7.2 + 1.2
  return 7.2 + 5.1 + 4.5 * (floor - 2) + 1.2

function factoryTypeEfficiency(t):
  if t.type == 'light-steel': return 2.1
  return t.floors
```

function validateInput(projectData, selectedProducts, productOptions):
  errors = []

  // 基本参数校验
  if projectData.landArea <= 0: errors.push('用地面积必须大于0')
  if projectData.far <= 0: errors.push('容积率必须大于0')
  if projectData.heightLimit <= 0: errors.push('限高必须大于0')
  if projectData.ancillaryRatio < 0: errors.push('配套占比不能为负')

  // 配套用房选择校验
  if projectData.ancillaryRatio > 0:
    if !selectedProducts.has('dorm') and !selectedProducts.has('support'):
      errors.push('配套用房未选择：R2>0时必须选择配套楼或配套宿舍')

    totalAncillary = projectData.landArea * projectData.far * projectData.ancillaryRatio
    if totalAncillary > 2400 and selectedProducts.has('support') and !selectedProducts.has('dorm'):
      errors.push('配套用房面积未用尽，请勾选配套宿舍')

  // 厂房类型组合校验
  if selectedProducts.has('layer') and selectedProducts.has('light-steel') and !selectedProducts.has('split'):
    errors.push('未选择分栋厂房：选择分层厂房和轻钢厂房时必须选择分栋厂房')

  // 层数类型数量校验
  if selectedProducts.has('split') and productOptions['split']?.floors?.length > 2:
    errors.push('分栋厂房层数类型超过2种')
  if selectedProducts.has('layer') and productOptions['layer']?.floors?.length > 2:
    errors.push('分层厂房层数类型超过2种')

  // 面积段选择校验
  for id in selectedProducts:
    if id in ['light-steel', 'split', 'layer', 'tower', 'support']:
      if !productOptions[id]?.areas or productOptions[id].areas.length == 0:
        errors.push('请为「' + getProductName(id) + '」选择面积段')
    if id in ['split', 'layer']:
      if !productOptions[id]?.floors or productOptions[id].floors.length == 0:
        errors.push('请为「' + getProductName(id) + '」选择层数')

  return errors

## 三、第零步：前置处理（与第七轮一致）

```js
function step0_preprocess(selectedProducts, productOptions, Hl):
  layerFloors = []
  splitFloors = []
  downgradedFloors = []
  layerDowngraded = false
  errors = []

  // 0.1 分层厂房层数降级
  if selectedProducts.has('layer'):
    originalLayerFloors = productOptions['layer'].floors || []
    for floor in originalLayerFloors:
      currentFloor = floor
      totalH = computeTotalHeight('layer', currentFloor)

      while totalH > Hl and currentFloor > 1:
        currentFloor--
        totalH = computeTotalHeight('layer', currentFloor)

      if totalH <= Hl:
        if currentFloor >= 5:
          layerFloors.push(currentFloor)
          if currentFloor < floor: layerDowngraded = true
        else if currentFloor <= 4:
          downgradedFloors.push(currentFloor)
      else:
        errors.push('分层厂房 ' + floor + 'F 即使降到1层仍超限高')

    if layerFloors.length == 0:
      selectedProducts.delete('layer')

  // 0.2 分栋厂房层数降级
  if selectedProducts.has('split'):
    originalSplitFloors = productOptions['split'].floors || []
    for floor in originalSplitFloors:
      totalH = computeTotalHeight('split', floor)
      if totalH <= Hl:
        splitFloors.push(floor)
    if splitFloors.length == 0:
      selectedProducts.delete('split')

  // 0.3 分层降级为分栋的合并处理
  if downgradedFloors.length > 0:
    selectedProducts.delete('layer')
    if !selectedProducts.has('split'):
      selectedProducts.add('split')

    originalLayerAreas = productOptions['layer']?.areas || []
    existingSplitAreas = productOptions['split']?.areas || []
    mergedAreas = uniqueSort([...existingSplitAreas, ...originalLayerAreas])
    if !productOptions['split']:
      productOptions['split'] = { floors: [], areas: [] }
    productOptions['split'].areas = mergedAreas

    convertedFloors = uniqueSort(downgradedFloors)
    existingSplitFloors = productOptions['split'].floors || []
    mergedFloors = uniqueSort([...existingSplitFloors, ...convertedFloors])
    productOptions['split'].floors = mergedFloors

  // 0.4 更新 productOptions 中的分层参数
  if selectedProducts.has('layer') and layerFloors.length > 0:
    productOptions['layer'].floors = layerFloors

  // 0.5 配套宿舍参考高度上限
  referenceHeight = Hl
  if layerFloors.length > 0:
    referenceHeight = max(layerFloors.map(f => computeTotalHeight('layer', f)))

  return {
    selectedProducts,
    productOptions,
    layerFloors,
    splitFloors,
    referenceHeight,
    layerDowngraded,
    errors
  }
```

## 六、步骤三：厂房预处理（本轮核心修改）

### 6.1 辅助函数：按 FL 阈值分配栋数

```js
function allocateByFL(FL, areas):
  // areas 已按从大到小排序
  n = areas.length
  counts = array(n, 0)

  if FL <= 4:
    counts[0] = 1  // 最高面积段 1 栋
  else if FL <= 6:
    for i from 0 to n-1: counts[i] = 1  // 各面积段 1 栋
  else if FL <= 8:
    for i from 0 to n-2: counts[i] = 1
    counts[n-1] = 2  // 最小面积段 2 栋，其余 1 栋
  else:
    for i from 0 to n-1: counts[i] = 2  // 各面积段 2 栋

  return counts
```

### 6.3 主函数

```js
function step3_factory_preprocess(selectedProducts, productOptions, layerFloors, splitFloors, targetBase, targetCap, fixedBase, fixedCap):
  fixedConfigs = []
  usedBase = 0
  usedCap = 0

  hasLightSteel = selectedProducts.has('light-steel')
  hasSplit = selectedProducts.has('split')
  hasLayer = selectedProducts.has('layer')

  remainingCap = targetCap - fixedCap

  // ============================================================
  // 情况一：分栋 + 轻钢组合 → 方案 B
  // ============================================================
  if hasLightSteel and hasSplit:
    // 有分层时，先固定 1 栋最低层数最小面积分层
    if hasLayer and layerFloors.length > 0:
      minLayerFloor = min(layerFloors)
      minLayerArea = min(productOptions['layer'].areas)
      cfg = createFixedConfig('layer', minLayerFloor, minLayerArea, 1)
      fixedConfigs.push(cfg)
      usedBase += cfg.base
      usedCap += cfg.unitCap
      layerFloors = []  // 该分层即为全部分层
      remainingCap -= cfg.unitCap

    // 方案 B：对较低层数分栋做手动分配
    if splitFloors.length > 0:
      lowerFloor = min(splitFloors)
      higherFloors = splitFloors.filter(f => f != lowerFloor)
      areas = [...productOptions['split'].areas].sort((a, b) => b - a)
      maxArea = areas[0]
      a = maxArea * lowerFloor
      FL = remainingCap / a

      counts = allocateByFL(FL, areas)
      for i from 0 to areas.length - 1:
        if counts[i] > 0:
          cfg = createFixedConfig('split', lowerFloor, areas[i], counts[i])
          fixedConfigs.push(cfg)
          usedBase += cfg.base * cfg.count
          usedCap += cfg.unitCap * cfg.count
          remainingCap -= cfg.unitCap * cfg.count

      splitFloors = higherFloors  // 较低分栋退出

    remainingBase = targetBase - fixedBase - usedBase
    remainingCap = targetCap - fixedCap - usedCap

    // B1：只剩轻钢
    if splitFloors.length == 0:
      singleConfigs = createConfigs('light-steel', 2.1, productOptions)
      return {
        fixedConfigs,
        bhConfigs: [],
        blConfigs: [],
        singleConfigs,
        isSingleType: true,
        isCaseB: true,
        remainingBase,
        remainingCap
      }

    // B2：轻钢 + 较高层数分栋
    // 此时 splitFloors 应只剩 1 种层数
    higherFloor = splitFloors[0]
    splitConfigs = createConfigs('split', higherFloor, productOptions)
    lightSteelConfigs = createConfigs('light-steel', 2.1, productOptions)

    // 按实际平均效率决定哪一侧为 Bh/Bl
    splitEff = splitConfigs.length
      ? sum(splitConfigs.map(c => c.unitCap / c.base)) / splitConfigs.length
      : higherFloor
    lightSteelEff = lightSteelConfigs.length
      ? sum(lightSteelConfigs.map(c => c.unitCap / c.base)) / lightSteelConfigs.length
      : 2.1

    if lightSteelEff > splitEff:
      bhConfigs = lightSteelConfigs
      blConfigs = splitConfigs
    else:
      bhConfigs = splitConfigs
      blConfigs = lightSteelConfigs

    bhConfigs.forEach(c => c.side = 'bh')
    blConfigs.forEach(c => c.side = 'bl')

    return {
      fixedConfigs,
      bhConfigs,
      blConfigs,
      singleConfigs: [],
      isSingleType: false,
      isCaseB: true,
      isCaseB2: true,
      remainingBase,
      remainingCap,
      typeSideMap: buildTypeSideMap(bhConfigs, blConfigs)
    }

  // ============================================================
  // 情况二：四层分栋 + 六层分层，且无轻钢 → 方案 A
  // ============================================================
  if hasSplit and hasLayer and !hasLightSteel:
    if splitFloors.length == 1 and splitFloors[0] == 4 and
       layerFloors.length == 1 and layerFloors[0] == 6:
      bhConfigs = createConfigs('layer', 6, productOptions)
      bhConfigs.forEach(c => c.side = 'bh')
      blConfigs = createConfigs('split', 4, productOptions)
      blConfigs.forEach(c => c.side = 'bl')

      return {
        fixedConfigs,
        bhConfigs,
        blConfigs,
        singleConfigs: [],
        isSingleType: false,
        isCaseA: true,
        remainingBase: targetBase - fixedBase,
        remainingCap: targetCap - fixedCap,
        typeSideMap: buildTypeSideMap(bhConfigs, blConfigs)
      }

  // ============================================================
  // 情况三：通用 FL 预处理
  // ============================================================
  allTypes = []
  if hasLightSteel:
    allTypes.push({ type: 'light-steel', floors: 2.1 })
  if hasSplit and splitFloors.length > 0:
    for floor in splitFloors:
      allTypes.push({ type: 'split', floors: floor })
  if hasLayer and layerFloors.length > 0:
    for floor in layerFloors:
      allTypes.push({ type: 'layer', floors: floor })

  while allTypes.length > 2:
    allTypes.sort((a, b) => factoryTypeEfficiency(b) - factoryTypeEfficiency(a))
    highest = allTypes[0]

    if highest.type == 'light-steel':
      // 轻钢效率偏低，通常不会成为最高效率。
      // 若出现轻钢为最高效率（例如只选了轻钢+更低效产品），保留进入 Bh/Bl。
      break

    areas = [...productOptions[highest.type].areas].sort((a, b) => b - a)
    maxArea = areas[0]
    a = maxArea * highest.floors

    if highest.type == 'layer':
      FL = remainingCap / a / 2
    else:
      FL = remainingCap / a

    counts = allocateByFL(FL, areas)
    for i from 0 to areas.length - 1:
      if counts[i] > 0:
        cfg = createFixedConfig(highest.type, highest.floors, areas[i], counts[i])
        fixedConfigs.push(cfg)
        usedBase += cfg.base * cfg.count
        usedCap += cfg.unitCap * cfg.count
        remainingCap -= cfg.unitCap * cfg.count

    allTypes = allTypes.slice(1)  // 移除最高效率类型

  remainingBase = targetBase - fixedBase - usedBase
  remainingCap = targetCap - fixedCap - usedCap

  // ============================================================
  // 剩余类型创建 Bh/Bl 或单一层数
  // ============================================================
  if allTypes.length == 0:
    return {
      fixedConfigs,
      bhConfigs: [],
      blConfigs: [],
      singleConfigs: [],
      isSingleType: true,
      remainingBase,
      remainingCap,
      typeSideMap: {}
    }

  if allTypes.length == 1:
    t = allTypes[0]
    singleConfigs = createConfigs(t.type, t.floors, productOptions)
    return {
      fixedConfigs,
      bhConfigs: [],
      blConfigs: [],
      singleConfigs,
      isSingleType: true,
      remainingBase,
      remainingCap,
      typeSideMap: { [t.type + '_' + t.floors]: 'bh' }
    }

  // allTypes.length == 2
  t1 = allTypes[0]
  t2 = allTypes[1]
  eff1 = factoryTypeEfficiency(t1)
  eff2 = factoryTypeEfficiency(t2)

  if eff1 >= eff2:
    bhConfigs = createConfigs(t1.type, t1.floors, productOptions)
    blConfigs = createConfigs(t2.type, t2.floors, productOptions)
  else:
    bhConfigs = createConfigs(t2.type, t2.floors, productOptions)
    blConfigs = createConfigs(t1.type, t1.floors, productOptions)

  bhConfigs.forEach(c => c.side = 'bh')
  blConfigs.forEach(c => c.side = 'bl')

  return {
    fixedConfigs,
    bhConfigs,
    blConfigs,
    singleConfigs: [],
    isSingleType: false,
    remainingBase,
    remainingCap,
    typeSideMap: buildTypeSideMap(bhConfigs, blConfigs)
  }

function buildTypeSideMap(bhConfigs, blConfigs):
  map = {}
  for c in bhConfigs: map[c.id + '_' + c.floors] = 'bh'
  for c in blConfigs: map[c.id + '_' + c.floors] = 'bl'
  return map
```

### 7.2 密度调整版 Bh/Bl（用于方案 A 和方案 B2）

```js
function step4_core_algorithm_with_adjustment(bhConfigs, blConfigs, remainingBase, remainingCap, projectData, fixedBase, fixedCap, warningCollector):
  if bhConfigs.length == 0 and blConfigs.length == 0:
    return { counts: [], bhConfigs, blConfigs, isSingleType: true, typeSideMap: {} }

  if bhConfigs.length == 0:
    counts = solveSingleType(blConfigs, remainingCap, remainingBase)
    return {
      counts,
      bhConfigs: [],
      blConfigs,
      isSingleType: true,
      typeSideMap: buildTypeSideMap([], blConfigs),
      adjustedD: null
    }

  if blConfigs.length == 0:
    counts = solveSingleType(bhConfigs, remainingCap, remainingBase)
    return {
      counts,
      bhConfigs,
      blConfigs: [],
      isSingleType: true,
      typeSideMap: buildTypeSideMap(bhConfigs, []),
      adjustedD: null
    }

  S = projectData.landArea
  D = calcDensity(projectData.far)

  effBh = average(bhConfigs.map(c => c.unitCap / c.base))
  effBl = average(blConfigs.map(c => c.unitCap / c.base))
  avgEff = remainingCap / remainingBase
  EPS = 1e-9

  // 情况 1：avgEff 在区间内，正常 Bh/Bl
  if avgEff > effBl + EPS and avgEff < effBh - EPS:
    return runNormalBhBl(bhConfigs, blConfigs, remainingBase, remainingCap, D)

  // 情况 2：avgEff 低于 Bl → 需要提高 avgEff（降低密度 D）
  if avgEff <= effBl + EPS:
    adjustedD = D * 0.95
    newTargetBase = S * adjustedD
    newRemainingBase = newTargetBase - fixedBase
    newAvgEff = remainingCap / newRemainingBase

    if newAvgEff > effBl + EPS and newAvgEff < effBh - EPS:
      return runNormalBhBl(bhConfigs, blConfigs, newRemainingBase, remainingCap, adjustedD)
    else:
      // 仍不满足，抛弃 Bh（分层/较高分栋），Bl 走单一层数
      warningCollector.push('产品选择与容积率不匹配')
      counts = solveSingleType(blConfigs, remainingCap, newRemainingBase)
      return {
        counts,
        bhConfigs: [],
        blConfigs,
        isSingleType: true,
        typeSideMap: buildTypeSideMap([], blConfigs),
        adjustedD: adjustedD,
        discardedType: bhConfigs[0].id + '_' + bhConfigs[0].floors
      }

  // 情况 3：avgEff 高于 Bh → 需要降低 avgEff（提高密度 D）
  if avgEff >= effBh - EPS:
    adjustedD = D * 1.05
    newTargetBase = S * adjustedD
    newRemainingBase = newTargetBase - fixedBase
    newAvgEff = remainingCap / newRemainingBase

    if newAvgEff > effBl + EPS and newAvgEff < effBh - EPS:
      return runNormalBhBl(bhConfigs, blConfigs, newRemainingBase, remainingCap, adjustedD)
    else:
      // 仍不满足，抛弃 Bl（分栋/轻钢），Bh 走单一层数
      warningCollector.push('产品选择与容积率不匹配')
      counts = solveSingleType(bhConfigs, remainingCap, newRemainingBase)
      return {
        counts,
        bhConfigs,
        blConfigs: [],
        isSingleType: true,
        typeSideMap: buildTypeSideMap(bhConfigs, []),
        adjustedD: adjustedD,
        discardedType: blConfigs[0].id + '_' + blConfigs[0].floors
      }

function runNormalBhBl(bhConfigs, blConfigs, remainingBase, remainingCap, targetD):
  effBh = average(bhConfigs.map(c => c.unitCap / c.base))
  effBl = average(blConfigs.map(c => c.unitCap / c.base))
  avgEff = remainingCap / remainingBase

  x = (remainingCap - effBl * remainingBase) / (effBh - effBl)
  y = remainingBase - x
  targetCapBh = effBh * x
  targetCapBl = effBl * y
  targetBaseBh = x
  targetBaseBl = y

  bhResult = solveIntegerEquation(targetCapBh, bhConfigs, targetBaseBh * 1.05)
  blResult = solveIntegerEquation(targetCapBl, blConfigs, targetBaseBl * 1.05)

  finalCounts = [...bhResult.counts, ...blResult.counts]

  return {
    counts: finalCounts,
    bhConfigs,
    blConfigs,
    isSingleType: false,
    typeSideMap: buildTypeSideMap(bhConfigs, blConfigs),
    adjustedD: targetD
  }
```

### 7.4 步骤四入口

```js
function step4_dispatch(factoryResult, projectData, fixedBase, fixedCap, warningCollector):
  bhConfigs = factoryResult.bhConfigs
  blConfigs = factoryResult.blConfigs
  singleConfigs = factoryResult.singleConfigs
  remainingBase = factoryResult.remainingBase
  remainingCap = factoryResult.remainingCap

  // 单一层数
  if factoryResult.isSingleType:
    if singleConfigs.length == 0:
      return { counts: [], configs: [], isSingleType: true, typeSideMap: {} }
    return step4_single_layer(singleConfigs, remainingCap, remainingBase)

  // 方案 A 或方案 B2：不同产品类型，需要密度调整保两种产品
  if factoryResult.isCaseA or factoryResult.isCaseB2:
    return step4_core_algorithm_with_adjustment(
      bhConfigs, blConfigs, remainingBase, remainingCap,
      projectData, fixedBase, fixedCap, warningCollector
    )

  // 普通情况（同类型产品或通用 FL 预处理后的 2 种类型）
  return step4_core_algorithm(bhConfigs, blConfigs, remainingBase, remainingCap)
```
